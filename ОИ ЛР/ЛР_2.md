### 3. Текст программы

```c++
#include <iostream>

using namespace std;

int main () {
	const int N = 2;
	const int M = 2;
	const int K = 2;
	int A[N][M];
	int B[M][K];
	int C[N][K];
	//Задать матрицу А//
	for (int i=0; i<N; i++)
	{
		for (int j=0; j<M; j++)
		{
			cout << "A["<<i+1<<"], ["<<j+1<<"]=";
			cin >> A[i][j];
		}
	}
	//Задать матрицу B//
	for (int x=0; x<M; x++)
	{
		for (int y=0; y<K; y++)
		{
			cout <<"B["<<x+1<<"], ["<<y+1<<"]=";
			cin >> B[x][y];
		}
	}
	//Вывод матрицы А//
	for (int i=0; i<N; i++)
	{
		for (int j=0; j<M; j++)
		{
			cout << A[i][j] << " ";
		}
		cout << endl;
	}
	//Вывод матрицы B//
	for (int x=0; x<M; x++)
	{
		for (int y=0; y<K; y++)
		{
			cout << B[x][y] << " ";
		}
		cout << endl;
	}
	//Расчет матрицы С//
	for (int i=0; i<N; i++)
	{
		for (int j=0; j<K; j++)
		{
			C[i][j]=0;
			for (int k=0; k<M; k++)
			{
				C[i][j]+=A[i][k]*B[k][j];
			}
		}
	}
	//Вывод матрицы С//
	for (int i=0; i<N; i++)
	{
		for (int j=0; j<M; j++)
		{
			cout << C[i][j] << " ";
		}
		cout << endl;
	}
	system ("pause");
	return 0;
}
```
